## 视频问答处理  
```python
# --- 视频网课导师模式 ---

video_file_path = "lecture_1.mp4" # 确保文件名一致

if os.path.exists(video_file_path):
    print("正在处理视频文件（这涉及到视频帧采样，请稍候）...")
    # 上传并指定类型为 video/mp4
    video_upload = client.files.upload(path=video_file_path)

    # 等待处理（视频解析比 PDF 慢得多）
    while video_upload.state == "PROCESSING":
        print(".", end="")
        time.sleep(5)
        video_upload = client.files.get(name=video_upload.name)

    print("\n✅ 视频已就绪！")

    # 专门为“精准翻译”设计的 Prompt（提示词）
    # 这里使用了“角色定义”和“多步翻译策略”
    translation_prompt = """
    你是一位精通离散数学和计算机科学的专业翻译官。
    请观看提供的网课视频，并执行以下任务：
    1. 提取视频中的所有原始英文语音内容（Transcript）。
    2. 将其翻译成中文。要求：
       - 翻译要贴合中国学生的语言习惯，不要有“翻译腔”。
       - 所有的专业术语（如: Discrete Probability, Graph Theory, Adjacency Matrix）必须保留英文。
       - 使用 LaTeX 格式输出数学公式。
    3. 以“时间戳 - 翻译内容”的格式输出，并对核心知识点进行简要解释。
    """

    response = client.models.generate_content(
        model="gemini-2.5-pro",
        contents=[video_upload, translation_prompt]
    )

    print("\n" + "="*40 + "\n翻译与学习指南：\n")
    print(response.text)
```  

## 字幕翻译
```pthon  
# 单元格 2：字幕翻译专用（直接使用 Cell 1 定义好的 client）

# 1. 准备文件（确保你左侧文件夹已经上传了 lesson_1.srt）
input_file = "lesson_1.srt"
output_file = "translated_lesson_1.srt"

if not os.path.exists(input_file):
    print("❌ 错误：没找到 lesson_1.srt，请先上传！")
else:
    # 2. 读取内容
    with open(input_file, "r", encoding="utf-8") as f:
        srt_content = f.read()

    print("导师正在进行深度翻译与去重...")

    # 3. 直接调用 client（它就在内存里等着你）
    response = client.models.generate_content(
        model="gemini-2.5-pro",
        contents=[
            "你是一位离散数学专家翻译。请将这段冗余的 YouTube SRT 字幕去重并翻译为专业的中文。保留英文术语，严格保持 SRT 格式。",
            srt_content
        ]
    )

    # 4. 保存结果
    with open(output_file, "w", encoding="utf-8") as f:
        f.write(response.text)

    print(f"✅ 翻译完成！右键点击左侧 {output_file} 即可下载。")
```
