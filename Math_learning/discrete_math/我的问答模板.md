```python
video_file_path = "lecture_1.mp4" # 确保文件名一致

    请观看提供的网课视频，并执行以下任务：
    1. 提取视频中的所有原始英文语音内容（Transcript）。
    2. 将其翻译成中文。要求：
       - 翻译要贴合中国学生的语言习惯，不要有“翻译腔”。
       - 所有的专业术语（如: Discrete Probability, Graph Theory, Adjacency Matrix）必须保留英文。
       - 使用 LaTeX 格式输出数学公式。
    3. 以“时间戳 - 翻译内容”的格式输出，并对核心知识点进行简要解释。
    """

    response = client.models.generate_content(
        model="gemini-2.5-pro",

    print(response.text)
```  

## 字幕翻译
```python  
# 离散数学字幕“一键清道夫”翻译脚本
import google.genai as genai
from google.genai import types
import os

# 1. 仍然使用你那个 U3-o 的 Key
MY_API_KEY = "在这里粘贴你的API密钥" 
client = genai.Client(api_key=MY_API_KEY, http_options={'api_version': 'v1alpha'})

# 2. 读取你上传的那个“脏”字幕文件
# 请确保你在左侧文件夹上传了名为 lesson_1.srt 的文件
input_file = "lesson_1.srt"

if not os.path.exists(input_file):
    print("❌ 错误：请先在左侧上传 lesson_1.srt")
else:
    with open(input_file, "r", encoding="utf-8") as f:
        content = f.read()

    print("正在进行深度逻辑翻译（去重+专业对齐）...")
    
    # 3. 构造超级提示词 (Prompt Engineering)
    # 既然你是程序员，你应该能看出这里在定义“翻译算法”的逻辑
    instruction = """
    你是一位精通离散数学的资深教授和翻译专家。
    任务：处理这段 YouTube 自动生成的冗余 SRT 字幕。
    
    要求：
    1. 逻辑去重：原始文本有严重的滚动重复（Rolling Captions），请将重复的短语合并，每句话只翻译一次完整的语意。
    2. 术语对齐：必须使用《离散数学及其应用》(Rosen) 教材的中文术语。
       - Proposition -> 命题
       - Connective -> 联结词
       - Negation/Conjunction/Disjunction -> 否定/合取/析取
       - Truth Table -> 真值表
    3. 自然流畅：不要逐字硬翻，要像中国大学教授讲课的语气。
    4. 格式保持：严格返回 SRT 格式。如果合并了重复行，请把时间戳范围拉长。
    """

    response = client.models.generate_content(
        model="gemini-2.5-pro",
        contents=[instruction, content]
    )

    # 4. 输出文件到本地磁盘
    output_file = "translated_lesson_1.srt"
    with open(output_file, "w", encoding="utf-8") as f:
        f.write(response.text)

    print(f"✅ 翻译完成！文件已生成：{output_file}")
```
